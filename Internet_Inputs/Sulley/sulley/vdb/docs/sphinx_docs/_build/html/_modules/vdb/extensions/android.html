[docs]def adb(db, line):
    '''
    Pass a command directly to the adb bridge command.
    (mostly just so you don't need another prompt)

    Usage adb <adb args>
    '''
    argv = e_cli.splitargs(line)
    db.vprint( vt_android.adbCommand('adb', *argv) )
